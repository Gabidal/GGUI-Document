Button - Examples

As of 0.1.8 the Button class is removed, and replaced with just a textField with a borders and text to center.

#include "ggui.h" using namespace GGUI; int main() { GGUI::GGUI( // You want press this one, since it does not have an on_click handle node(new textField( position(STYLES::left) | // <-- anchor the text to the left side text("You Can't press me") | anchor(ANCHOR::CENTER) | // <-- make the text centered enable_border(true) )) | // This one you can press because of the on_click event present node(new textField( position(STYLES::right) | // <-- anchor the text to right side text("Press me") | anchor(ANCHOR::CENTER) | // <-- make the text centered enable_border(true) | on_click([](element*){ // <-- Self parameter return true; // <-- Stop propagation of event }) )) ); while (true) { // ... } }
